SELECT f.nom, COUNT(fi.ingredient_id) AS nb
FROM focaccia f
JOIN focaccia_ingredient fi ON fi.focaccia_id = f.id
GROUP BY f.id
ORDER BY nb DESC
LIMIT 1;
